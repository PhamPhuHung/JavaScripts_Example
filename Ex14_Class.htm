Example

Class template:

Original Function

class Car {
   constructor() {
     this.name = name;
     this.year = year;
   }
   age(x) {
     return x - this.year;
   }
}

let myCar = new Car("Ford", 2014);
let date = new Date();
let currentYear = date.getFullYear();
let str = "My car is " + myCar.age(currentYear) + " years old"; // My Car is 7 years old
