The addEventListener() method

Add an Event Handler to an Element

// Ad an onclick event
var e11_1 = document.getElementById("e11_1");
e11_1.addEventListener("click", function() { alert("Hello World"); });

// Ad an onmouseover event
var e11_2 = document.getElementById("e11_2");
e11_2.addEventListener("mouseover", function() { e11_2.innerHTML = "Thank you"; });
e11_2.addEventListener("mouseout", function() { e11_2.innerHTML = "Mouse over me"; });

Add many Event Handlers to the same Element

// Ad an onclick event
var e12_1 = document.getElementById("e12_1");
e12_1.addEventListener("click", function() { alert("Hello World 1st"); });
e12_1.addEventListener("click", function() { alert("Hello World 2nd"); });

Add an Event Handler to the window Object

var e13_1 = document.getElementById("e13_1");
window.addEventListener("resize", function() { e13_1.innerHTML = Math.random(); });

Event Bubbling or Event Capturing

Set the third parameter of addEventListener()

  • false (default): bubbling - inside to ouside
  • true: capturing - outside to inside
var e15_1_div = document.getElementById("e15_1_div");
var e15_1_p = document.getElementById("e15_1_p");
var e15_2_div = document.getElementById("e15_2_div");
var e15_2_p = document.getElementById("e15_2_p");
e15_1_div.addEventListener("click", function(){ alert("You clicked the orange element");}, false);
e15_1_p.addEventListener("click", function(){ alert("You clicked the white element");}, false);
e15_2_div.addEventListener("click", function(){ alert("You clicked the orange element");}, true);
e15_2_p.addEventListener("click", function(){ alert("You clicked the white element");}, true);
